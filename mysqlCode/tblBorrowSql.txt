CREATE TABLE IF NOT EXISTS `borrow` (
 `id` INT NOT NULL AUTO_INCREMENT,
 `bookisbn` VARCHAR(100) NOT NULL,
 `borrowedOn` DATE NOT NULL,
 `returnedOn` DATE NOT NULL,
 `userid` INT NOT NULL,
 PRIMARY KEY (`id`),
 UNIQUE INDEX `id_UNIQUE` (`id` ASC) VISIBLE,
 INDEX `index_user` (`userid` ASC) INVISIBLE,
 INDEX `index_isbn` (`bookisbn` ASC) INVISIBLE,
 CONSTRAINT `isbn_fk` FOREIGN KEY (`bookisbn`) REFERENCES `books` (`isbn`) ON DELETE NO ACTION ON UPDATE NO ACTION,
 CONSTRAINT `user_fk` FOREIGN KEY (`userid`) REFERENCES `users` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION
);
